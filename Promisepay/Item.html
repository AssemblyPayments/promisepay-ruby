def fees(options = {})
  response = JSON.parse(@client.get("items/#{send(:id)}/fees", options).body)
  fees = response.key?('fees') ? response['fees'] : []
  fees.map { |attributes| Promisepay::Fee.new(@client, attributes) }
end
